import flask from flask import Flask, render_template import requests import json app = Flask(__name__) @app.route('/') def index(): return render_template('index.html') @app.route('/submit', methods=['POST']) def submit(): link = flask.request.form['link'] # Query the large language model response = requests.get('http://example.com/api/query?link=' + link) data = json.loads(response.text) # Render the relevant information from the language model in an organized and readable format on the left side of the screen return render_template('index.html', link=link, data=data) if __name__ == '__main__': app.run() #### index.html #### PDF/Webpage Viewer

{% endif %} {% endif %}
